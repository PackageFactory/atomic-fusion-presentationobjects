declare(strict_types=1);

namespace Vendor\Default\Presentation\Component\Crossing;

enum TrafficLight:int
{
    case LIGHT_RED = 1;
    case LIGHT_YELLOW = 2;
    case LIGHT_GREEN = 3;

    public function getIsRed(): bool
    {
        return $this === self::LIGHT_RED;
    }

    public function getIsYellow(): bool
    {
        return $this === self::LIGHT_YELLOW;
    }

    public function getIsGreen(): bool
    {
        return $this === self::LIGHT_GREEN;
    }
}
